113   /** 
114    * (smt)2 replaying object.
115    * This Object is private. Methods are cited but not documented.
116    */
117   var smtRep = {
118     i:       0,                           // mouse tracking global counter var
119     j:       1,                           // registration points size global counter var
120     jMax:    0,                           // registration points size limit
121     play:    null,                        // mouse tracking identifier
122     jg:      null,                        // canvas area for drawing
123     jgClust: null,                        // layer for clustering
124     jgHelper: null,                       // layer for displaying text info
125     viewport: { width:0, height:0 },      // data normalization

128     /** 
129      * Create drawing canvas layer.
130      */
131     createCanvas: function(layerName) 
132     {
133       // canvas layer for mouse trackig
134       var jg = document.createElement("div");
135           jg.id             = layerName;
136           jg.style.position = "absolute";
137           jg.style.top      = 0;
138           jg.style.left     = 0;
139           jg.style.width    = 100 + '%';
140           jg.style.height   = 100 + '%';
141           jg.style.zIndex   = aux.getNextHighestDepth() + 1;
142       
143       // helper layer for text
144       var jgHelp = document.createElement("div");
145           jgHelp.id              = layerName + "Help";
146           jgHelp.style.zIndex    = jg.style.zIndex + 1;
147           
148       // layer for clustering
149       var opacity = 40;
150       var jgClust = document.createElement("div");
151           jgClust.id              = layerName + "Clust";
152           jgClust.style.position  = "absolute";
153           jgClust.style.top       = 0;
154           jgClust.style.left      = 0;
155           jgClust.style.width     = 100 + '%';
156           jgClust.style.height    = 100 + '%';
157           jgClust.style.opacity   = opacity/100; // for W3C browsers
158           jgClust.style.filter    = "alpha(opacity="+opacity+")"; // only for IE
159           jgClust.style.zIndex    = jg.style.zIndex + 2;
160           
161       var body  = document.getElementsByTagName("body")[0];
162           body.appendChild(jg);
163           body.appendChild(jgHelp);
164           body.appendChild(jgClust);
165           
166       // set the canvas areas for drawing both mouse tracking and clustering
167       smtRep.jg = new jsGraphics(jg.id);
168       smtRep.jgHelper = new jsGraphics(jgHelp.id);
169       smtRep.jgClust = new jsGraphics(jgClust.id);
170     },

172      * Create background layer.

175     {
176       var opacity = 50, // background layer opacity (%)
177           // set layer above the mouse tracking one
178           jg = document.getElementById(layerName),
179           doc = aux.getPageSize();
180           
181       var bg = document.createElement("div");
182           bg.id                     = layerName + "Bg";
183           bg.style.position         = "absolute";
184           bg.style.top              = 0;
185           bg.style.left             = 0;
186           bg.style.width            = doc.width + 'px'; 
187           bg.style.height           = doc.height + 'px';
188           bg.style.overflow         = "hidden";
189           bg.style.backgroundColor  = smtOpt.bgColor;
190           bg.style.opacity          = opacity/100; // for W3C browsers
191           bg.style.filter           = "alpha(opacity="+opacity+")"; // only for IE
192           bg.style.zIndex           = jg.style.zIndex - 1;
193       
194       var body  = document.getElementsByTagName("body")[0];
195           body.appendChild(bg);
196     },
197     /** 
198      * Draw line.
199      */
200     drawLine: function(ini,end) 
201     {
202         smtRep.jg.setColor(smtOpt.regLn);
203         smtRep.jg.drawLine(ini.x,ini.y, end.x,end.y);
204         smtRep.jg.paint();
205     },
206     /** 
207      * Draw mouse click.
208      */
209     drawClick: function(x,y, isDragAndDrop) 
210     {
211       var size;
212       if (!isDragAndDrop) {
213         size = 12;
214         var offset = 3;
215         smtRep.jg.setColor(smtOpt.click);
216         smtRep.jg.drawLine(x-size,y-size, x-offset,y-offset);
217         smtRep.jg.drawLine(x-size,y+size, x-offset,y+offset);
218         smtRep.jg.drawLine(x+size,y-size, x+offset,y-offset);
219         smtRep.jg.drawLine(x+size,y+size, x+offset,y+offset);
220       } else {
221         size = 6;
222         smtRep.jg.setColor(smtOpt.dDrop);
223         smtRep.jg.drawRect(x-size/2,y-size/2, size,size);
224       }
225       smtRep.jg.paint();
226     },
227     /** 
228      * Draw direction arrow in a line.
229      */
230     drawDirectionArrow: function(ini,end)
231     {
232       var a = ini.x - end.x,
233           b = ini.y - end.y,
234           s = 4;
235       if (a>0 && b>0) {
236         smtRep.jg.drawPolyline([end.x-s,end.x,end.x+s], [end.y+s,end.y,end.y+s]);
237       } else if (a<0 && b>0) {
238         smtRep.jg.drawPolyline([end.x-s,end.x,end.x-s], [end.y-s,end.y,end.y+s]);
239       } else if (a<0 && b<0) {
240         smtRep.jg.drawPolyline([end.x-s,end.x,end.x+s], [end.y-s,end.y,end.y-s]);
241       } else if (a>0 && b<0) {
242         smtRep.jg.drawPolyline([end.x+s,end.x,end.x+s], [end.y-s,end.y,end.y+s]);
243       }
244       smtRep.jg.paint();
245     },
246     /** 
247      * Draw mouse cursor.
248      */
249     drawCursor: function(x,y, color) 
250     {
251       smtRep.jg.setColor(color);
252       smtRep.jg.fillPolygon([x,x,   x+4, x+6, x+9, x+7, x+15], 
253                             [y,y+15,y+15,y+23,y+23,y+15,y+15]);
254       smtRep.jg.paint();
255     },
256     /** 
257      * Draw registration point.
258      */
259     drawRegistrationPoint: function(x,y) 
260     {
261       smtRep.jg.setColor(smtOpt.regPt);
262       smtRep.jg.fillRect(x-1, y-1, 3, 3);
263       smtRep.jg.paint();
264     },
265     /** 
266      * Draw time-depending circle.
267      */
268     drawVariableCircle: function(x,y, size) 
269     {
270       // use multiplier to normalize all circles: 0 < norm < 1
271       var norm = aux.roundTo(size/smtRep.jMax); 
272       if (size * norm === 0 ) { return; }
273       // limit size to 1/2 of current window width (px)
274       if (size > smtData.wcurr/2) { size = Math.round(smtData.wcurr/2 * norm); }
275       // draw
276       smtRep.jg.setColor(smtOpt.varCir);
277       smtRep.jg.drawEllipse(x - size/2, y - size/2, size, size);
278       smtRep.jg.paint();
279     },
280     /** 
281      * Gives a visual clue when the user is not using the mouse.
282      */
283     drawMouseStop: function(x,y) 
284     {
285       if (!smtOpt.realTime) { return; }
286       
287       var fontSize   = 16,
288           circleSize = 50;
289       smtRep.jgHelper.setColor(smtOpt.varCir);
290       smtRep.jgHelper.fillEllipse(x - circleSize/2, y - circleSize/2, circleSize, circleSize);
291       smtRep.jgHelper.setColor("black");
292       smtRep.jgHelper.setFont("sans-serif", fontSize+'px', Font.BOLD);
293       // center the text in vertical 
294       smtRep.jgHelper.drawString("stopped...", x, y - fontSize/2);
295       smtRep.jgHelper.paint();
296     },   
297     /** 
298      * Draw centroid as a star.
299      */
300     drawCentroid: function()
301     {
302       smtRep.jg.setColor(smtOpt.cenPt);
303       // the centroid is computed discarding null distances
304       var u = Math.round(aux.array.sum(xclean) * smtRep.discrepance.x / xclean.length),
305           v = Math.round(aux.array.sum(yclean) * smtRep.discrepance.y / yclean.length),
306           l = 20; // centroid line length
307       smtRep.jg.setStroke(5);
308     	smtRep.jg.drawLine(u, v, u+l, v-l); // 1st quadrant
309     	smtRep.jg.drawLine(u, v, u-l, v-l); // 2nd quadrant
310     	smtRep.jg.drawLine(u, v, u-l, v+l); // 3rd quadrant
311     	smtRep.jg.drawLine(u, v, u+l, v+l); // 4th quadrant
312     	smtRep.jg.setStroke(0); // reset strokes
313     	smtRep.jg.paint();
314     },
315     /** 
316      * Draw cluster as a circle.
317      */
318     drawClusters: function() 
319     {
320       smtRep.jgClust.setColor(smtOpt.clust);
321       var size;
322       for (var i = 0, clusters = smtData.clustsize.length; i < clusters; ++i) {
323         size = smtData.clustsize[i];
324         smtRep.jgClust.fillEllipse(smtData.xclusters[i] * smtRep.discrepance.x - size/2, smtData.yclusters[i] * smtRep.discrepance.y - size/2, size, size);
325       }
326       smtRep.jgClust.paint();
327     },
328     /** 
329      * Get euclidean distance from point a to point b.
330      */
331     distance: function(a,b) 
332     {
333       return Math.sqrt( Math.pow(a.x - b.x,2) + Math.pow(a.y - b.y,2) );
334     },
335     /** 
336      * Auto scrolls the browser window.
337      */
338     checkAutoScrolling: function(x,y) 
339     {
340       if (!smtOpt.realTime) { return; }
341       // center current mouse coords on the viewport
342       aux.doScroll({xpos:x, ypos:y, width:smtRep.viewport.width, height:smtRep.viewport.height});
343     },
344     /** 
345      * (smt)2 realtime drawing algorithm.
346      */
347     playMouse: function() 
348     {
349       if (smtRep.paused) { return; }
350 
351       // mouse coords normalization
352       var iniMouse = { 
353                         x: smtData.xcoords[smtRep.i] * smtRep.discrepance.x,
354                         y: smtData.ycoords[smtRep.i] * smtRep.discrepance.y 
355                      };
356       var endMouse = { 
357                         x: smtData.xcoords[smtRep.i+1] * smtRep.discrepance.x,
358                         y: smtData.ycoords[smtRep.i+1] * smtRep.discrepance.y 
359                      };
360       var iniClick = {
361                         x: smtData.xclicks[smtRep.i] * smtRep.discrepance.x, 
362                         y: smtData.yclicks[smtRep.i] * smtRep.discrepance.y
363                      };
364       var endClick = {
365                         x: smtData.xclicks[smtRep.i+1] * smtRep.discrepance.x, 
366                         y: smtData.yclicks[smtRep.i+1] * smtRep.discrepance.y
367                      };
368       
369       // draw entry point
370       if (smtRep.i === 0) {
371         smtRep.drawCursor(iniMouse.x,iniMouse.y, smtOpt.entryPt);
372       }
373       
374       // main loop to draw mouse trail
375       if (smtRep.i < smtData.xcoords.length) 
376       {
377         var mouseDistance = smtRep.distance(iniMouse,endMouse);
378         // draw registration points
379         if (mouseDistance) {
380           // there is mouse movement
381           if (!smtOpt.dirVect) {
382             // show static squares
383             smtRep.drawRegistrationPoint(iniMouse.x,iniMouse.y);
384           } else {
385             // show direction pseudo-arrows
386             smtRep.drawDirectionArrow(iniMouse,endMouse);
387           }
388           // variable circles
389           if (smtRep.j > 1) {
390             smtRep.drawVariableCircle(iniMouse.x, iniMouse.y, smtRep.j);
391             smtRep.jgHelper.clear();
392           }
393           // reset variable circles size
394           smtRep.j = 1;
395         } else {
396           // mouse stop: store variable size (circles)
397           ++smtRep.j;
398           smtRep.drawMouseStop(iniMouse.x, iniMouse.y);
399         }
400         // draw lines
401         smtRep.drawLine(iniMouse,endMouse);
402         // draw clicks
403         if (iniClick.x) {
404           var clickDistance = smtRep.distance(iniClick,endClick);
405           if (!clickDistance) {
406             // is a single click
407             smtRep.drawClick(endClick.x,endClick.y, false);
408           } else if (endClick.x !== 0) {
409             // is drag and drop
410             smtRep.drawClick(iniClick.x,iniClick.y, true);
411           }
412         }
413         // update mouse coordinates
414         ++smtRep.i;
415         // check auto scrolling
416         smtRep.checkAutoScrolling(endMouse.x, endMouse.y);
417     	}
418       
419       // draw exit point
420       else {
421     	  // rewind count 1 step to access the last mouse coordinate
422     	  --smtRep.i;
423     	  iniMouse.x = smtData.xcoords[smtRep.i] * smtRep.discrepance.x;
424         iniMouse.y = smtData.ycoords[smtRep.i] * smtRep.discrepance.y;
425         // draw exit point
426     	  smtRep.drawCursor(iniMouse.x,iniMouse.y, smtOpt.exitPt);
427     	  // draw clusters
428     	  smtRep.drawClusters();
429         // draw centroid (average mouse position) 
430         smtRep.drawCentroid();
431         // clear mouse tracking
432         clearInterval(smtRep.play);
433         smtRep.play = null;
434         smtRep.jgHelper.clear();
435         // load next trail
436         aux.loadNextMouseTrail(smtData);
437     	}
438     },
439     /** 
440      * Replay method: static or dynamic.
441      */
442     replay: function(realtime) 
443     {
444       if (realtime) {
445         // fps are stored in smtData object, so we can use that value here
446         var interval = Math.round(1000/smtData.fps);
447         smtRep.play = setInterval(smtRep.playMouse, interval);
448       } else {
449         // static mouse tracking visualization 
450         for (var k = 0, total = smtData.xcoords.length; k <= total; ++k) {
451           smtRep.playMouse();
452         }
453       }
454     },
455     /** 
456      * Reload method: mouse tracking layers are redrawn.
457      */
458     reset: function() 
459     {
460       clearInterval(smtRep.play);
461       smtRep.paused = false;
462       // reset counters
463       smtRep.i = 0;
464       smtRep.j = 1;    
465       // clear canvas  
466       smtRep.jg.clear();
467       smtRep.jgClust.clear();
468     },
469     /** 
470      * User can pause the mouse replay by pressing the SPACE key, 
471      * or toggle replay mode by pressing the ESC key.
472      */
473     helpKeys: function(e) 
474     {
475       // use helpKeys only in realtime replaying
476       if (!smtOpt.realTime) { return; }
477       
478       if (!e) { e = window.event; }
479       var code = e.keyCode || e.which;
480       // on press ESC key finish drawing
481       if (code == 27) {
482         // clear main loop
483         clearInterval(smtRep.play);
484         smtRep.paused = false;
485         // set this flag
486         smtOpt.realTime = false;
487         // end drawing from the current position
488         for (var k = smtRep.i, total = smtData.xcoords.length; k <= total; ++k) {
489           smtRep.playMouse();
490         }
491       } else if (code == 32) {
492         // on press space bar toggle drawing
493         smtRep.paused = !smtRep.paused;
494       }
495     },
496     /** 
497      * System initialization.
498      */
499     init: function() 
500     {
501       var vp = aux.getWindowSize();
502       smtRep.viewport.width = vp.width;
503       smtRep.viewport.height = vp.height;
504       
505       // compute the discrepance ratio
506       if (smtData.wprev && smtData.hprev) {
507         smtRep.discrepance.x = aux.roundTo(smtData.wcurr / smtData.wprev);
508         smtRep.discrepance.y = aux.roundTo(smtData.hcurr / smtData.hprev);  
509       }
510       //aux.log(smtRep.discrepance.x+" x "+smtRep.discrepance.y+" | "+smtRep.viewport.width+" x "+smtRep.viewport.height);
511           
512       // precalculate the user stops: useful for time-depending circles and path centroid
513       var stops = [];      
514       var size = 1;
515       for (var k = 0, len = smtData.xcoords.length; k < len; ++k) {
516         if (smtData.xcoords[k] == smtData.xcoords[k+1] && smtData.ycoords[k] == smtData.ycoords[k+1]) {
517           ++size;
518         } else {
519           // store all user stops (time) for drawing variable circles later
520           if (size > 1) { stops.push(size); }
521           size = 1;
522           // store clean mouse coordinates
523           xclean.push(smtData.xcoords[k]);
524           yclean.push(smtData.ycoords[k]);
525         }
526       }
527       // set max size for variable circles
528       smtRep.jMax = aux.array.max(stops);       
529       // common suffix for tracking canvas and background layers
530       var smtName = "smtCanvas";
531       // set the canvas layer
532       smtRep.createCanvas(smtName);
533       // draw the background layer
534       if (smtOpt.bgLayer) { smtRep.setBgCanvas(smtName); }
535       // allow mouse replay over Flash animations
536       aux.allowTrackingOnFlashObjects();
537       // init
538       smtRep.replay(smtOpt.realTime);
539     }
540   };
