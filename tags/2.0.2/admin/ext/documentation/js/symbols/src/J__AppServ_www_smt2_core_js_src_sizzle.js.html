 16 // optimization where it does not always call our comparision
 17 // function. If that is the case, discard the hasDuplicate value.

 24 var Sizzle = function(selector, context, results, seed) {
 25 	results = results || [];
 26 	var origContext = context = context || document;
 27 
 28 	if ( context.nodeType !== 1 && context.nodeType !== 9 ) {
 29 		return [];
 30 	}
 31 	
 32 	if ( !selector || typeof selector !== "string" ) {
 33 		return results;
 34 	}
 35 
 36 	var parts = [], m, set, checkSet, extra, prune = true, contextXML = isXML(context),
 37 		soFar = selector;
 38 	
 39 	// Reset the position of the chunker regexp (start from head)
 40 	while ( (chunker.exec(""), m = chunker.exec(soFar)) !== null ) {
 41 		soFar = m[3];
 42 		
 43 		parts.push( m[1] );
 44 		
 45 		if ( m[2] ) {
 46 			extra = m[3];
 47 			break;
 48 		}
 49 	}
 50 
 51 	if ( parts.length > 1 && origPOS.exec( selector ) ) {
 52 		if ( parts.length === 2 && Expr.relative[ parts[0] ] ) {
 53 			set = posProcess( parts[0] + parts[1], context );
 54 		} else {
 55 			set = Expr.relative[ parts[0] ] ?
 56 				[ context ] :
 57 				Sizzle( parts.shift(), context );
 58 
 59 			while ( parts.length ) {
 60 				selector = parts.shift();
 61 
 62 				if ( Expr.relative[ selector ] ) {
 63 					selector += parts.shift();
 64 				}
 65 				
 66 				set = posProcess( selector, set );
 67 			}
 68 		}
 69 	} else {
 70 		// Take a shortcut and set the context if the root selector is an ID
 71 		// (but not if it'll be faster if the inner selector is an ID)
 72 		if ( !seed && parts.length > 1 && context.nodeType === 9 && !contextXML &&
 73 				Expr.match.ID.test(parts[0]) && !Expr.match.ID.test(parts[parts.length - 1]) ) {
 74 			var ret = Sizzle.find( parts.shift(), context, contextXML );
 75 			context = ret.expr ? Sizzle.filter( ret.expr, ret.set )[0] : ret.set[0];
 76 		}
 77 
 78 		if ( context ) {
 79 			var ret = seed ?
 80 				{ expr: parts.pop(), set: makeArray(seed) } :
 81 				Sizzle.find( parts.pop(), parts.length === 1 && (parts[0] === "~" || parts[0] === "+") && context.parentNode ? context.parentNode : context, contextXML );
 82 			set = ret.expr ? Sizzle.filter( ret.expr, ret.set ) : ret.set;
 83 
 84 			if ( parts.length > 0 ) {
 85 				checkSet = makeArray(set);
 86 			} else {
 87 				prune = false;
 88 			}
 89 
 90 			while ( parts.length ) {
 91 				var cur = parts.pop(), pop = cur;
 92 
 93 				if ( !Expr.relative[ cur ] ) {
 94 					cur = "";
 95 				} else {
 96 					pop = parts.pop();
 97 				}
 98 
 99 				if ( pop == null ) {
100 					pop = context;
101 				}
102 
103 				Expr.relative[ cur ]( checkSet, pop, contextXML );
104 			}
105 		} else {
106 			checkSet = parts = [];
107 		}
108 	}
109 
110 	if ( !checkSet ) {
111 		checkSet = set;
112 	}
113 
114 	if ( !checkSet ) {
115 		Sizzle.error( cur || selector );
116 	}
117 
118 	if ( toString.call(checkSet) === "[object Array]" ) {
119 		if ( !prune ) {
120 			results.push.apply( results, checkSet );
121 		} else if ( context && context.nodeType === 1 ) {
122 			for ( var i = 0; checkSet[i] != null; i++ ) {
123 				if ( checkSet[i] && (checkSet[i] === true || checkSet[i].nodeType === 1 && contains(context, checkSet[i])) ) {
124 					results.push( set[i] );
125 				}
126 			}
127 		} else {
128 			for ( var i = 0; checkSet[i] != null; i++ ) {
129 				if ( checkSet[i] && checkSet[i].nodeType === 1 ) {
130 					results.push( set[i] );
131 				}
132 			}
133 		}
134 	} else {
135 		makeArray( checkSet, results );
136 	}
137 
138 	if ( extra ) {
139 		Sizzle( extra, origContext, results, seed );
140 		Sizzle.uniqueSort( results );
141 	}
142 
143 	return results;
144 };

146 Sizzle.uniqueSort = function(results){
147 	if ( sortOrder ) {
148 		hasDuplicate = baseHasDuplicate;
149 		results.sort(sortOrder);
150 
151 		if ( hasDuplicate ) {
152 			for ( var i = 1; i < results.length; i++ ) {
153 				if ( results[i] === results[i-1] ) {
154 					results.splice(i--, 1);
155 				}
156 			}
157 		}
158 	}
159 
160 	return results;
161 };

167 Sizzle.find = function(expr, context, isXML){
168 	var set, match;
169 
170 	if ( !expr ) {
171 		return [];
172 	}
173 
174 	for ( var i = 0, l = Expr.order.length; i < l; i++ ) {
175 		var type = Expr.order[i], match;
176 		
177 		if ( (match = Expr.leftMatch[ type ].exec( expr )) ) {
178 			var left = match[1];
179 			match.splice(1,1);
180 
181 			if ( left.substr( left.length - 1 ) !== "\\" ) {
182 				match[1] = (match[1] || "").replace(/\\/g, "");
183 				set = Expr.find[ type ]( match, context, isXML );
184 				if ( set != null ) {
185 					expr = expr.replace( Expr.match[ type ], "" );
186 					break;
187 				}
188 			}
189 		}
190 	}
191 
192 	if ( !set ) {
193 		set = context.getElementsByTagName("*");
194 	}
195 
196 	return {set: set, expr: expr};
197 };

199 Sizzle.filter = function(expr, set, inplace, not){
200 	var old = expr, result = [], curLoop = set, match, anyFound,
201 		isXMLFilter = set && set[0] && isXML(set[0]);
202 
203 	while ( expr && set.length ) {
204 		for ( var type in Expr.filter ) {
205 			if ( (match = Expr.leftMatch[ type ].exec( expr )) != null && match[2] ) {
206 				var filter = Expr.filter[ type ], found, item, left = match[1];
207 				anyFound = false;
208 
209 				match.splice(1,1);
210 
211 				if ( left.substr( left.length - 1 ) === "\\" ) {
212 					continue;
213 				}
214 
215 				if ( curLoop === result ) {
216 					result = [];
217 				}
218 
219 				if ( Expr.preFilter[ type ] ) {
220 					match = Expr.preFilter[ type ]( match, curLoop, inplace, result, not, isXMLFilter );
221 
222 					if ( !match ) {
223 						anyFound = found = true;
224 					} else if ( match === true ) {
225 						continue;
226 					}
227 				}
228 
229 				if ( match ) {
230 					for ( var i = 0; (item = curLoop[i]) != null; i++ ) {
231 						if ( item ) {
232 							found = filter( item, match, i, curLoop );
233 							var pass = not ^ !!found;
234 
235 							if ( inplace && found != null ) {
236 								if ( pass ) {
237 									anyFound = true;
238 								} else {
239 									curLoop[i] = false;
240 								}
241 							} else if ( pass ) {
242 								result.push( item );
243 								anyFound = true;
244 							}
245 						}
246 					}
247 				}
248 
249 				if ( found !== undefined ) {
250 					if ( !inplace ) {
251 						curLoop = result;
252 					}
253 
254 					expr = expr.replace( Expr.match[ type ], "" );
255 
256 					if ( !anyFound ) {
257 						return [];
258 					}
259 
260 					break;
261 				}
262 			}
263 		}
264 
265 		// Improper expression
266 		if ( expr === old ) {
267 			if ( anyFound == null ) {
268 				Sizzle.error( expr );
269 			} else {
270 				break;
271 			}
272 		}
273 
274 		old = expr;
275 	}
276 
277 	return curLoop;
278 };
